递归回溯求解

```JavaScript
function isvalid(arr,depth){
    for(let i=0;i<depth;i++){
        if(arr[depth]===arr[i]) return false
        if(Math.abs(arr[depth]-arr[i])===Math.abs(i-depth)) return false
    }
    return true
}
let res=[],result=[]
let tmp
function travel(depth,n){
    if(depth>n){
        tmp=res.slice()
        result.push(tmp)
    }
    else{
        for(let i=1;i<=n;i++){
            res[depth-1]=i
            if(isvalid(res,depth-1)){
                travel(depth+1,n)
            }
        }
    }
}
```
